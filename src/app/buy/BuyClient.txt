- FE에러 처리방법
useEffect(() => {
	if (!isError || !error) return;

	if (error instanceof ApiError) {
		if (error.code === "HOLD_EXPIRED") {
			const returnUrl = (error.data as { returnUrl?: string })?.returnUrl || "/cart";
			alert("상품 점유 시간이 만료되었습니다.");
			router.replace(returnUrl);
			return;
		}

		if (error.code === "HOLD_NOT_FOUND") {
			alert("잘못된 접근입니다.");
			router.replace("/cart");
			return;
		}
	}

	alert("구매 정보를 불러오지 못했습니다.");
	router.replace("/cart");
}, [isError, error, router]);
